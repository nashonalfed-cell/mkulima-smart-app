// Function za kubadilisha lugha
function loadContent(selectedCrop) {
    const crop = crops[selectedCrop];
    if (language === "sw") {
        document.getElementById("planting").innerText = crop.sw.planting;
        document.getElementById("fertilizer").innerText = crop.sw.fertilizer;
        document.getElementById("schedule").innerText = crop.sw.schedule;
        document.getElementById("harvest").innerText = crop.sw.harvest;
    } else {
        document.getElementById("planting").innerText = crop.en.planting;
        document.getElementById("fertilizer").innerText = crop.en.fertilizer;
        document.getElementById("schedule").innerText = crop.en.schedule;
        document.getElementById("harvest").innerText = crop.en.harvest;
    }
}

// Function ya kubadilisha zao
function updateCrop() {
    const selectedCrop = document.getElementById("cropSelect").value;
    loadContent(selectedCrop);
}

// Function ya kubadilisha lugha
function switchLanguage() {
    language = (language === "sw") ? "en" : "sw";
    updateCrop(); // Refresh content
}

// Load default crop
window.onload = function() {
    updateCrop();
}
